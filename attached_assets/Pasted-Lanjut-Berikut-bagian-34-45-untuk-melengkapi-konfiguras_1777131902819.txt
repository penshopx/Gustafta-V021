# 34. Conversation State Machine

```yaml id="ixgejt"
state_machine:
  initial_state: onboarding

  states:
    onboarding:
      description: "Menyambut user dan mengarahkan ke menu utama."
      on:
        user_selects_role_lookup: role_lookup
        user_selects_standard_lookup: standard_lookup
        user_selects_kkni_lookup: kkni_lookup
        user_selects_recommendation: role_discovery
        user_selects_learning: learning_menu
        user_selects_assessment: assessment_start
        user_selects_scenario: scenario_menu
        user_selects_checklist: evidence_checklist
        user_selects_glossary: glossary_lookup

    role_lookup:
      description: "Mencari jabatan Mekanikal berdasarkan input user."
      on:
        exact_role_found: role_detail
        multiple_roles_found: role_selection
        no_role_found: fallback_role_not_found
        role_ambiguous: role_clarification

    role_detail:
      description: "Menampilkan detail jabatan kerja."
      on:
        user_requests_assessment: role_assessment
        user_requests_learning: related_learning_module
        user_requests_checklist: evidence_checklist
        user_requests_standard: standard_detail
        user_requests_comparison: compare_roles
        user_returns_menu: main_menu

    kkni_lookup:
      description: "Mencari jabatan berdasarkan jenjang KKNI."
      on:
        kkni_valid: show_roles_by_kkni
        kkni_invalid: kkni_validation_error
        user_selects_role: role_detail

    standard_lookup:
      description: "Mencari jabatan berdasarkan kode SKKNI/SKKK atau standar kompetensi."
      on:
        standard_found: standard_detail
        multiple_standards_found: standard_selection
        standard_not_found: fallback_standard_not_found

    role_discovery:
      description: "Memetakan rekomendasi jabatan berdasarkan pengalaman user."
      on:
        profile_complete: role_recommendation
        answer_ambiguous: role_discovery_clarification
        user_selects_area: area_recommendation

    learning_menu:
      description: "Menu modul pembelajaran Mekanikal."
      on:
        module_selected: learning_module
        module_complete: module_quiz
        quiz_complete: learning_recommendation
        user_requests_scenario: scenario_menu
        user_requests_assessment: assessment_start

    assessment_start:
      description: "Memulai asesmen mandiri."
      on:
        assessment_area_selected: assessment_questions
        assessment_complete: assessment_result
        user_requests_recommendation: adaptive_learning

    scenario_menu:
      description: "Latihan studi kasus Mekanikal."
      on:
        scenario_selected: scenario_prompt
        user_submits_answer: scenario_feedback
        user_requests_model_answer: scenario_model_answer
        user_requests_next_scenario: scenario_menu

    mock_interview:
      description: "Simulasi wawancara asesmen."
      on:
        question_asked: user_answer_wait
        user_submits_answer: mock_feedback
        continue_interview: next_mock_question
        finish_interview: session_summary

    evidence_checklist:
      description: "Membuat checklist bukti kompetensi."
      on:
        area_selected: generate_area_checklist
        role_selected: generate_role_checklist
        user_requests_detail: checklist_detail
        user_requests_mock_interview: mock_interview

    glossary_lookup:
      description: "Menjelaskan istilah SKK/SKKNI/Mekanikal."
      on:
        term_found: glossary_response
        term_not_found: fallback_glossary

    fallback:
      description: "Menangani input ambigu atau tidak ditemukan."
      on:
        user_selects_topic: corresponding_state
        repeated_failure: main_menu
```

learning_module:
  module_id: lifting_sling_inspection

  target_area: "Crane, Lifting, Slinging, dan Rigging"

  field_example: >
    Sebelum pengangkatan material baja, operator menemukan sling aus dan label

    - "Dapat mencatat temuan dalam checklist pemeriksaan."

  quiz:
    question: "Apa tindakan terbaik saat sling terlihat aus dan label kapasitas tidak terbaca?"
    options:

      B: "Digunakan dengan mengurangi beban"
      C: "Menghentikan pekerjaan sementara dan mengganti sling"
      D: "Melanjutkan pekerjaan tanpa dokumentasi"
    correct_answer: "C"
    explanation: >
      Sling yang tidak layak harus diganti. Pekerjaan lifting tidak boleh dilanjutkan
      sampai alat bantu angkat aman dan sesuai kapasitas.
