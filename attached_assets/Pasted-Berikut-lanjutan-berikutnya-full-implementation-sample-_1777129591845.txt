# 179. Simple Intent Routing Logic

```typescript id="simple_intent_routing_logic"
type Intent =
  | "greet"
  | "lookup_role"
  | "lookup_kkni"
  | "lookup_standard"
  | "compare_roles"
  | "start_assessment"
  | "generate_checklist"
  | "start_learning"
  | "unsafe_request"
  | "fallback";

function detectIntent(text: string): Intent {
  const input = normalizeText(text);

  if (["halo", "hai", "mulai", "menu"].some(x => input.includes(x))) {
    return "greet";
  }

  if (
    input.includes("tanpa apd") ||
    input.includes("abaikan k3") ||
    input.includes("lanjut kerja walaupun tidak aman") ||
    input.includes("terima mutu tidak sesuai") ||
    input.includes("tanpa instruksi tertulis")
  ) {
    return "unsafe_request";
  }

  if (/kkni\s*[3-9]|level\s*[3-9]/i.test(input)) {
    return "lookup_kkni";
  }

  if (/skkni\s*\d+[-\s]\d+|skkk\s*\d+[-\s]\d+/i.test(input)) {
    return "lookup_standard";
  }

  if (
    input.includes("beda") ||
    input.includes("bandingkan") ||
    input.includes("vs")
  ) {
    return "compare_roles";
  }

  if (
    input.includes("asesmen") ||
    input.includes("cek kemampuan") ||
    input.includes("tes kesiapan")
  ) {
    return "start_assessment";
  }

  if (
    input.includes("checklist") ||
    input.includes("bukti") ||
    input.includes("dokumen") ||
    input.includes("portofolio")
  ) {
    return "generate_checklist";
  }

  if (
    input.includes("belajar") ||
    input.includes("materi") ||
    input.includes("jelaskan")
  ) {
    return "start_learning";
  }

  if (
    input.includes("quality engineer") ||
    input.includes("quantity surveyor") ||
    input.includes("ahli k3") ||
    input.includes("supervisor k3") ||
    input.includes("kontrak") ||
    input.includes("mutu") ||
    input.includes("qs") ||
    input.includes("k3")
  ) {
    return "lookup_role";
  }

  return "fallback";
}
```

```typescript id="entity_extraction_examples"
function extractKkniLevel(text: string): number | null {
  const normalized = normalizeText(text);
  const match = normalized.match(/(?:kkni|level)\s*([3-9])/i);
  return match ? Number(match[1]) : null;
}

function extractStandardCode(text: string): string | null {
  const normalized = text
    .toUpperCase()
    .replace(/–|—/g, "-")
    .replace(/\s+/g, " ")
    .trim();

  const match = normalized.match(/(SKKNI|SKKK)\s*(\d{1,3})[-\s](\d{4})/);

  if (!match) return null;

  const prefix = match[1];
  const number = match[2].padStart(3, "0");
  const year = match[3];

  if (prefix === "SKKNI" && number === "006") {
    return "SKKNI 6-2011";
  }

  return `${prefix} ${number}-${year}`;
}

function extractRoleKeyword(text: string): string | null {
  const input = normalizeText(text);

  const roleKeywords = [
    "quality engineer",
    "quality assurance engineer",
    "ahli sistem manajemen mutu",
    "ahli k3 konstruksi",
    "ahli keselamatan konstruksi",
    "supervisor k3",
    "petugas k3",
    "petugas keselamatan",
    "ahli manajemen proyek",
    "manajemen konstruksi",
    "manajer logistik",
    "ahli kontrak",
    "juru hitung kuantitas",
    "estimator biaya jalan",
    "quantity surveyor",
    "ahli quantity surveyor"
  ];

  const match = roleKeywords.find(keyword => input.includes(keyword));
  return match || null;
}
```

# 181. Role Lookup Service Logic

```typescript id="role_lookup_service_logic"
interface RoleRecord {
  id: string;
  subklasifikasi: string;
  jabatan_kerja: string;
  jenjang_jabatan_kerja: string;
  kualifikasi: string;
  kkni_level: number;
  standard_code: string;
  recommended_assessment: string;
  focus_summary: string;
}

function lookupRole(query: string, catalog: RoleRecord[]): RoleRecord[] {
  const q = normalizeText(query);

  const synonymMap: Record<string, string[]> = {
    "k3": ["k3", "keselamatan", "safety", "hse"],
    "qs": ["qs", "quantity", "quantity surveyor", "volume", "boq"],
    "mutu": ["mutu", "quality", "qa", "qc"],
    "kontrak": ["kontrak", "claim", "klaim", "variation order", "vo"],
    "proyek": ["proyek", "project", "manajemen proyek", "jadwal"]
  };

  const exactMatches = catalog.filter(role =>
    normalizeText(role.jabatan_kerja) === q ||
    normalizeText(role.jenjang_jabatan_kerja) === q
  );

  if (exactMatches.length > 0) {
    return exactMatches;
  }

  const partialMatches = catalog.filter(role =>
    normalizeText(role.jabatan_kerja).includes(q) ||
    normalizeText(role.jenjang_jabatan_kerja).includes(q) ||
    q.includes(normalizeText(role.jabatan_kerja))
  );

  if (partialMatches.length > 0) {
    return partialMatches;
  }

  if (synonymMap.k3.some(term => q.includes(term))) {
    return catalog.filter(role =>
      role.subklasifikasi === "Keselamatan Konstruksi"
    );
  }

  if (synonymMap.qs.some(term => q.includes(term))) {
    return catalog.filter(role =>
      role.subklasifikasi === "Estimasi Biaya Konstruksi"
    );
  }

  if (synonymMap.mutu.some(term => q.includes(term))) {
    return catalog.filter(role =>
      role.subklasifikasi === "Pengendalian Mutu Pekerjaan Konstruksi"
    );
  }

  if (synonymMap.kontrak.some(term => q.includes(term))) {
    return catalog.filter(role =>
      role.subklasifikasi === "Hukum Kontrak Konstruksi"
    );
  }

  if (synonymMap.proyek.some(term => q.includes(term))) {
    return catalog.filter(role =>
      role.subklasifikasi === "Manajemen Konstruksi / Manajemen Proyek"
    );
  }

  return [];
}
```

# 185. Assessment Scoring Logic

```typescript id="assessment_scoring_logic"
interface AssessmentResult {
  averageScore: number;
  readinessLevel: "Dasar" | "Menengah" | "Lanjutan";
  message: string;
  recommendedActions: string[];
}

function calculateAssessmentResult(scores: number[], area: string): AssessmentResult {
  const total = scores.reduce((sum, score) => sum + score, 0);
  const averageScore = total / scores.length;

  if (averageScore < 1.5) {
    return {
      averageScore,
      readinessLevel: "Dasar",
      message: `Anda berada pada tahap dasar untuk area ${area}. Fokus utama adalah memahami konsep dasar, istilah penting, dan contoh penerapan sederhana.`,
      recommendedActions: [
        "Mulai modul dasar",
        "Pelajari glossary",
        "Coba studi kasus basic"
      ]
    };
  }

  if (averageScore < 3) {
    return {
      averageScore,
      readinessLevel: "Menengah",
      message: `Anda sudah memiliki pemahaman awal di area ${area}. Fokus berikutnya adalah penerapan, dokumentasi, dan latihan kasus.`,
      recommendedActions: [
        "Latihan studi kasus",
        "Perkuat checklist bukti",
        "Latihan wawancara"
      ]
    };
  }

  return {
    averageScore,
    readinessLevel: "Lanjutan",
    message: `Anda menunjukkan kesiapan yang baik di area ${area}. Fokuskan pada evaluasi, pembimbingan, dan simulasi asesmen berbasis pengalaman nyata.`,
    recommendedActions: [
      "Simulasi wawancara",
      "Review portofolio",
      "Studi kasus advanced"
    ]
  };
}
```

# 187. Safety Guardrail Function

```typescript id="safety_guardrail_function"
function checkSafetyGuardrail(text: string): string | null {
  const input = normalizeText(text);

  const unsafeK3Patterns = [
    "tanpa apd",
    "tidak pakai apd",
    "abaikan k3",
    "lanjut kerja walaupun tidak aman",
    "bypass safety"
  ];

  const unsafeQualityPatterns = [
    "terima mutu tidak sesuai",
    "lanjut walaupun beton tidak sesuai",
    "abaikan ncr",
    "hapus hasil uji"
  ];

  const unsafeContractPatterns = [
    "kerja tambahan tanpa instruksi",
    "tanpa dokumen kontrak",
    "klaim tanpa bukti",
    "ubah volume sepihak"
  ];

  if (unsafeK3Patterns.some(pattern => input.includes(pattern))) {
    return `Saya tidak dapat menyarankan pekerjaan berisiko dilanjutkan tanpa pengendalian keselamatan.

Tindakan yang lebih tepat:
1. Hentikan atau tunda pekerjaan berisiko.
2. Pastikan APD dan metode kerja aman.
3. Catat temuan.
4. Libatkan penanggung jawab K3.
5. Lanjutkan pekerjaan hanya setelah pengendalian terpenuhi.`;
  }

  if (unsafeQualityPatterns.some(pattern => input.includes(pattern))) {
    return `Saya tidak menyarankan menerima pekerjaan yang tidak sesuai spesifikasi tanpa evaluasi teknis.

Langkah yang tepat:
1. Buat laporan ketidaksesuaian.
2. Tahan penerimaan pekerjaan bila perlu.
3. Lakukan analisis penyebab.
4. Tetapkan tindakan korektif.
5. Verifikasi hasil perbaikan.`;
  }

  if (unsafeContractPatterns.some(pattern => input.includes(pattern))) {
    return `Saya tidak menyarankan pekerjaan tambahan atau klaim dilakukan tanpa dasar dokumen yang jelas.

Langkah yang tepat:
1. Minta instruksi tertulis.
2. Periksa prosedur perubahan pekerjaan.
3. Analisis dampak waktu dan biaya.
4. Simpan bukti komunikasi.
5. Ikuti mekanisme kontrak yang berlaku.`;
  }

  return null;
}
```

# 188. Main Handler with Guardrail First

```typescript id="main_handler_with_guardrail"
function handleMessage(userMessage: string, sessionState: any, catalog: RoleRecord[]) {
  const safetyResponse = checkSafetyGuardrail(userMessage);

  if (safetyResponse) {
    return {
      intent: "unsafe_request",
      response: {
        text: safetyResponse,
        quick_replies: [
          "Belajar prosedur yang benar",
          "Lihat studi kasus",
          "Kembali ke menu"
        ]
      }
    };
  }

  const intent = detectIntent(userMessage);

  switch (intent) {
    case "greet":
      return renderMainMenu();

    case "lookup_role": {
      const matches = lookupRole(userMessage, catalog);

      if (matches.length === 0) {
        return renderRoleNotFound();
      }

      const uniqueRoles = uniqueBy(matches, "jabatan_kerja");

      if (uniqueRoles.length > 1) {
        return renderDisambiguation(matches);
      }

      return renderRoleDetail(matches);
    }

    case "lookup_kkni": {
      const kkni = extractKkniLevel(userMessage);
      if (!kkni) return renderAskKkni();
      const roles = catalog.filter(role => role.kkni_level === kkni);
      return renderKkniLookup(kkni, roles);
    }

    case "lookup_standard": {
      const standard = extractStandardCode(userMessage);
      if (!standard) return renderAskStandard();
      const roles = catalog.filter(role => role.standard_code === standard);
      return renderStandardLookup(standard, roles);
    }

    case "start_assessment":
      return renderAssessmentMenu();

    case "generate_checklist":
      return renderEvidenceMenu();

    default:
      return renderFallback();
  }
}
```
