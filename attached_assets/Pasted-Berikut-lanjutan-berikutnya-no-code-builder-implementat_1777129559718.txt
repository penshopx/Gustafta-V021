```yaml id="no_code_assessment_node_pattern"
assessment_node_pattern:
  flow_id: "assessment_{{area_slug}}"
  node_sequence:
    - start_node: "node_assessment_{{area_slug}}_start"
    - question_nodes:
        - "node_assessment_{{area_slug}}_q1"

        - "node_assessment_{{area_slug}}_q4"

    - result_node: "node_assessment_{{area_slug}}_result"

  scoring:
    score_variable: "{{area_slug}}_score_total"
    answer_values:
      "0 - Belum memahami": 0
      "1 - Paham teori": 1
      "2 - Pernah menerapkan dengan arahan": 2
      "3 - Mampu mandiri": 3
      "4 - Mampu mengevaluasi/membimbing": 4

  result_logic:
    - condition: "average_score < 1.5"
      result: "Dasar"
    - condition: "average_score >= 1.5 and average_score < 3"
      result: "Menengah"
    - condition: "average_score >= 3"
      result: "Lanjutan"
```

# 171. Function Calling Tool Specs

```yaml id="function_calling_specs"
function_tools:

  - name: "lookup_role"
    description: "Cari detail jabatan kerja berdasarkan nama jabatan atau kata kunci."
    input_schema:
      type: object
      properties:
        query:
          type: string
          description: "Nama jabatan atau kata kunci, misalnya Quality Engineer."
      required:
        - query
    output_schema:
      type: object
      properties:
        matches:
          type: array
          items:
            type: object
            properties:
              role_id: { type: string }
              subklasifikasi: { type: string }
              jabatan_kerja: { type: string }
              jenjang_jabatan_kerja: { type: string }
              kualifikasi: { type: string }
              kkni_level: { type: integer }
              standard_code: { type: string }

  - name: "lookup_kkni"
    description: "Cari jabatan kerja berdasarkan jenjang KKNI."
    input_schema:
      type: object
      properties:
        kkni_level:
          type: integer
          enum: [3, 4, 5, 6, 7, 8, 9]
      required:
        - kkni_level

  - name: "lookup_standard"
    description: "Cari jabatan kerja berdasarkan kode SKKNI atau SKKK."
    input_schema:
      type: object
      properties:
        standard_code:
          type: string
          description: "Contoh: SKKNI 333-2013 atau SKKK 035-2022."
      required:
        - standard_code

  - name: "start_assessment"
    description: "Mulai asesmen mandiri berdasarkan area kompetensi."
    input_schema:
      type: object
      properties:
        area:
          type: string
          enum:
            - "k3"
            - "project"
            - "quality"
            - "contract"
            - "cost_qs"
      required:
        - area

  - name: "generate_evidence_checklist"
    description: "Buat checklist bukti kompetensi berdasarkan jabatan atau area."
    input_schema:
      type: object
      properties:
        target_role:
          type: string
        area:
          type: string
      required: []
```

# 172. Function Calling Response Policy

```yaml id="function_calling_response_policy"
function_response_policy:
  lookup_role:
    if_zero_matches:
      response: >
        Saya belum menemukan jabatan tersebut. Coba gunakan kata kunci umum
        seperti K3, Quality Engineer, Manajemen Proyek, Kontrak, atau QS.
    if_one_match:
      response: "Render role detail template."
    if_multiple_matches:
      response: "Render disambiguation list."

  lookup_kkni:
    response: >
      Kelompokkan hasil berdasarkan subklasifikasi. Jangan tampilkan seluruh
      official link kecuali pengguna meminta.

  lookup_standard:
    response: >
      Tampilkan semua jabatan yang memakai standar tersebut, urutkan berdasarkan
      KKNI dari rendah ke tinggi.

  start_assessment:
    response: >
      Tampilkan intro asesmen, disclaimer, pertanyaan pertama, dan skala 0–4.

  generate_evidence_checklist:
    response: >
      Kelompokkan bukti menjadi: pengalaman kerja, dokumen teknis, dokumen
      pengendalian, komunikasi, dan hasil pekerjaan.
```

# 176. Final Notes for Handoff to Builder Team

```text id="handoff_to_builder_team"
Catatan untuk tim builder:

1. Gunakan katalog jabatan sebagai database utama.
2. Jangan simpan data jabatan hanya di prompt; simpan dalam tabel agar mudah diperbarui.
3. Buat node statis untuk menu utama dan subklasifikasi.
4. Gunakan lookup dinamis untuk nama jabatan, KKNI, dan standar.
5. Untuk input singkat seperti “K3”, “QS”, “mutu”, tampilkan pilihan, jangan menebak satu jabatan.
6. Untuk asesmen MVP, gunakan 5 pertanyaan per area terlebih dahulu.
7. Untuk hasil asesmen, gunakan kategori Dasar, Menengah, dan Lanjutan.
8. Selalu tampilkan disclaimer bahwa hasil asesmen bukan keputusan resmi sertifikasi.
9. Tambahkan fallback tiga tingkat.
10. Uji minimal 20 contoh percakapan sebelum soft launch.
```
