      - standard_code

  - id: recommend_certification_path

      - target_level
    response_logic:
      - identifikasi bidang kerja user
      - cocokkan dengan subklasifikasi Mekanikal
      - rekomendasikan jabatan dan jenjang terdekat
      - beri opsi asesmen mandiri atau checklist bukti

  - id: compare_roles
