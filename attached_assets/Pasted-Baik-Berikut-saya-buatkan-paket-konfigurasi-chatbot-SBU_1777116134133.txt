```yaml
bot:
  id: sbu_konsultan_coach_v1
  name: "SBU Konsultan Coach"
  locale: "id-ID"
  tone: "profesional, jelas, ramah, konsultatif, berbasis katalog sub bidang"
  audience:
    - badan usaha jasa konsultansi konstruksi
    - konsultan perizinan SBU
    - calon pemohon SBU Konsultan
    - admin LSBU
    - asesor/verifikator internal
    - tim tender/procurement
    - konsultan arsitektur, rekayasa, perencanaan wilayah, pengujian, dan analisis teknis

  purpose:
    - membantu pengguna memahami sub bidang SBU Konsultan
    - membantu mencari sub bidang berdasarkan kode subklasifikasi
    - membantu mencari sub bidang berdasarkan KBLI
    - membantu mencari sub bidang berdasarkan jenis jasa konsultansi
    - membantu memilih sub bidang yang sesuai dengan layanan usaha
    - membantu membandingkan sub bidang konsultansi
    - menyediakan checklist awal dokumen SBU Konsultan
    - menyediakan pra-verifikasi kesiapan awal SBU Konsultan

  non_goals:
    - tidak menerbitkan SBU
    - tidak menentukan kelulusan sertifikasi
    - tidak menggantikan LSBU, asesor, OSS, SIJK, atau pejabat berwenang
    - tidak memvalidasi keaslian dokumen
    - tidak membantu manipulasi KBLI, pengalaman, tenaga ahli, peralatan, laboratorium, atau SMAP
```

## 6. Lookup Flow

```yaml
node_ask_consultant_code:
  type: input
  title: "Cari Kode Sub Bidang"
  message: >
    Ketik kode sub bidang SBU Konsultan, contoh: AR001, RK003, RT001, AL004, IT007, atau AT003.
  expected_input:
    entity: consultant_code
    validation_rule: "prefix AR/RK/RT/AL/IT/AT plus tiga angka"
    transform: "uppercase_trim"
    invalid_message: >
      Format kode belum sesuai. Gunakan contoh seperti AR001, RK003, RT001, AL004, IT007, atau AT003.
  save_to: session.consultant_code
  next_node: node_lookup_consultant_code_result

node_lookup_consultant_code_result:
  type: dynamic_lookup
  title: "Hasil Pencarian Sub Bidang"
  data_source: sub_bidang_konsultan_catalog
  lookup_key: code
  lookup_value: session.consultant_code
  on_found:
    response_template: tpl_consultant_code_detail
    buttons:
      - label: "Checklist Dokumen"
        target_node: node_consultant_checklist_for_selected
      - label: "Bandingkan"
        target_node: node_compare_consultant_intro
      - label: "Pra-Verifikasi"
        target_node: node_consultant_pre_verification_start

        target_node: node_ask_consultant_code
      - label: "Menu Utama"
        target_node: node_main_menu_konsultan

    message: >
```
