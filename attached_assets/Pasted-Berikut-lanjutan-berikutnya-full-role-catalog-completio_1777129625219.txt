# 198. Recommended Focus Summary Generator

```yaml id="focus_summary_generator"
focus_summary_generator:
  purpose: >
    Menghasilkan ringkasan fokus kompetensi otomatis berdasarkan subklasifikasi
    dan jabatan kerja.

  rules:

    if_area_k3:
      base_summary: >
        Fokus pada keselamatan konstruksi/K3, identifikasi bahaya,
        pengendalian risiko, pemeriksaan kepatuhan, pelaporan kondisi tidak
        aman, investigasi kejadian, dan pembinaan budaya keselamatan.

    if_area_project:
      base_summary: >
        Fokus pada perencanaan pelaksanaan, koordinasi sumber daya,
        pengendalian waktu, biaya, mutu, risiko, logistik, komunikasi,
        dan pelaporan progres proyek.

    if_area_quality:
      base_summary: >
        Fokus pada pengendalian mutu, inspeksi pekerjaan, verifikasi
        spesifikasi, dokumentasi mutu, NCR, tindakan korektif, audit,
        dan peningkatan sistem kualitas.

    if_area_contract:
      base_summary: >
        Fokus pada pemahaman kontrak, variation order, klaim, adendum,
        sengketa, pembuktian, hak dan kewajiban para pihak, serta risiko
        kontraktual proyek.

    if_area_cost_qs:
      base_summary: >
        Fokus pada pengukuran volume, BOQ, backup quantity, estimasi biaya,
        analisis harga satuan, progress payment, deviasi volume/biaya,
        dan evaluasi komersial proyek.

  seniority_adjustment:
    operator:
      add: >
        Penekanan pada pelaksanaan prosedur dasar dan pelaporan kepada atasan.

    teknisi_analis:
      add: >
        Penekanan pada penerapan teknis, pemeriksaan, analisis awal, supervisi,
        dan dokumentasi hasil pekerjaan.

    ahli:
      add: >
        Penekanan pada analisis kompleks, evaluasi, pengambilan keputusan,
        pengendalian sistem, dan pembinaan tim.
```
