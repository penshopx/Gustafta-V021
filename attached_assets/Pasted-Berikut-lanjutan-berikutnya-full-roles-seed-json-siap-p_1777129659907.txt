# 211. Validation Query for Complete Catalog

```sql id="validation_query_complete_catalog"
-- Jumlah total record jenjang jabatan
SELECT COUNT(*) AS total_role_levels
FROM skk_roles
WHERE classification_code = 'E';

-- Jumlah per subklasifikasi
SELECT
  subklasifikasi,
  COUNT(*) AS total_levels
FROM skk_roles
WHERE classification_code = 'E'
GROUP BY subklasifikasi
ORDER BY subklasifikasi;

-- Cek apakah semua record punya assessment mapping
SELECT *
FROM skk_roles
WHERE recommended_assessment IS NULL
   OR recommended_assessment = '';

-- Cek apakah semua KKNI berada dalam rentang 3-9
SELECT *
FROM skk_roles
WHERE kkni_level < 3
   OR kkni_level > 9;
```
